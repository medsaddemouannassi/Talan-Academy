fonction triSelection(ref T :tableau[1..N] d'entiers) :vide;
  var i,j,indice_cle :entier;
  début
    pour i allant de 1 à N-1 faire
      indice_cle = minimumSoustableau(T,i,N);
      echange(T[i],T[indice_cle]);
    finpour
  fin
finfonction
