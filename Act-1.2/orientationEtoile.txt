Activité 1.2.5 : Orientation avec des étoiles

début

var T1: tableau
T1 = [Sirius, Rigel, Bételgeuse, Arcturus, Aldébaran, Véga, Deneb, et Altaïr]

fonction triBulle(ref T :tableau[1..N] de chaines de caractères) : vide;
var i,j,k :entier;
début
pour i allant de 1 à N-1 faire
	pour j allant de N à i+1 par pas de -1 faire
		répéter
			si T[j][k] > T[j-1][k] alors
			echange(T, j, j-1);
			k <-- min(long(T[j]),long(T[j-1]))
			sinon si T[j][k] = T[j-1][k] alors
			k <-- k + 1	
			finsi
		jusqu'à k = min(long(T[j]),long(T[j-1]))
	finpour
finpour
	retourner(T)
fin
finfonction

fonction retrier() : ;
var étoile, réponse: chaine de caractère
début
répéter
afficher("voulez vous ajouter une étoile ? répondre par oui ou non")
saisir(réponse)
	si réponse = oui alors
	saisir(étoile)
	T1[long(T1)] = étoile
	triBulle(T1);
jusqu'à réponse = non
retourner(T1);
fin
finfonction

triBulle(T1);

fonction cherche(ref T :tableau[1..N] de chaines de caractères; val e :entier) :chaine de caractères;  
  var d,f :entier;
  fonction chercheRec(ref T :tableau[1..N] de chaines de caractères; val d,f,e :entier) :chaine de caractères;
  var m;
    début
      si f == d alors
         si T[d] == e alors
           retourner(f)
        sinon
            retourner(NUL)
        finsi
      sinon
        m = partieEntiere((d+f)/2);
        si T[m] < e alors
          retourner(chercheRec(T,m+1,f,e))
        sinon
          retourner(chercheRec(T,d,m,e))
        finsi
      finsi
    fin
  finfonction
  début
    d = 1;
    f = N;
    retourner(chercheRec(T,d,f,e))
  fin
finfonction

cherche(T1)

fin
